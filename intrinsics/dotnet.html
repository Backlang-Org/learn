Dotnet Intrinsics

Intrinsic Explanation Example
ld_null Load Null 
ld_null();
ld_str Load String 
ld_str("Hello World");
ldc_i4 Load Integer Constant 
ldc_i4(42);
ldc_i8 Load Long Integer Constant 
ldc_i8(420000);
ldc_r4 Load Float Constant 
ldc_r4(3.14);
ldc_r8 Load Long Float Constant 
ldc_r8(3.1400000011);
ldarg Load Argument 
ldarg(0);
ldloc Load Local 
ldloc(myVariable);
ldfld Load Field 
ldfld(myField);
stfld Store Field 
stfld(myField);
add Add 2 Values Together 
add();
neg Negate Value 
negate();
nop Do Nothing 
nop();
ret Return 
return();
throw Throw An Exception 
throw(Exception::new("Error"));
